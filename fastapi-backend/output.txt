**.NET Backend (C#)**
---------------------

### Dependencies

### Code

**Startup.cs**
```csharp
using Microsoft.AspNetCore.Authentication;
using Microsoft.AspNetCore.Authentication.Google;
using Microsoft.AspNetCore.Builder;
using Microsoft.AspNetCore.Hosting;
using Microsoft.Extensions.DependencyInjection;
using Microsoft.Extensions.Logging;

public class Startup
{
    public void ConfigureServices(IServiceCollection services)
    {
        services.AddAuthentication(options =>
        {
            options.DefaultScheme = "Google";
            options.DefaultChallengeScheme = "Google";
        })
        .AddGoogle(options =>
        {
            options.ClientId = "YOUR_CLIENT_ID";
            options.ClientSecret = "YOUR_CLIENT_SECRET";
        });

        services.AddSignalR();
    }

    public void Configure(IApplicationBuilder app, IWebHostEnvironment env)
    {
        app.UseRouting();
        app.UseAuthentication();
        app.UseEndpoints(endpoints =>
        {
            endpoints.MapHub<ChatHub>("/chat");
        });
    }
}
```

**React Frontend**

### Dependencies

* React 17.0.2
* `react-router-dom`

* `axios`

### Code

**App.js**
```jsx
import React, { useState, useEffect } from 'react';
import { BrowserRouter, Route, Switch } from 'react-router-dom';
import { Button, TextField, Grid } from '@material-ui/core';
import axios from 'axios';

function App() {
  const [user, setUser] = useState(null);
  const [message, setMessage] = useState('');
  const [messages, setMessages] = useState([]);

  useEffect(() => {
    const token = localStorage.getItem('token');
    if (token) {
      axios.get('/api/user', {
        headers: {
          Authorization: `Bearer ${token}`,
        },
      })
      .then(response => {
        setUser(response.data);
      })
      .catch(error => {
        console.error(error);
      });
    }
  }, []);

  const handleLogin = () => {
    window.location.href = '/login';
  };

  const handleSendMessage = () => {
    axios.post('/api/chat/send-message', {
      user: user.name,
      text: message,
    })
    .then(response => {
      setMessages([...messages, response.data]);
      setMessage('');
    })
    .catch(error => {
      console.error(error);
    });
  };

  return (
    <BrowserRouter>
      <Switch>
        <Route path="/login" component={Login} />
        <Route path="/" render={() => (
          <div>
            {user ? (
              <div>
                <Grid container spacing={2}>
                  <Grid item xs={12}>
                    <TextField
                      label="Message"
                      value={message}
                      onChange={e => setMessage(e.target.value)}
                    />
                  </Grid>
                  <Grid item xs={12}>
                    <Button variant="contained" onClick={handleSendMessage}>
                      Send
                    </Button>
                  </Grid>
                  <Grid item xs={12}>
                    <ul>
                      {messages.map((msg, index) => (
                        <li key={index}>{msg.user}: {msg.text}</li>
                      ))}
                    </ul>
                  </Grid>
                </Grid>
              </div>
            ) : (
              <Button variant="contained" onClick={handleLogin}>
                Login with Google
              </Button>
            )}
          </div>
        )} />
      </Switch>
    </BrowserRouter>
  );
}

export default App;
```

```javascript
// app/controllers/TodoController.js
const Todo = require('../models/Todo');

class TodoController {
    async getTodos(req, res) {
        const todos = await Todo.find().exec();
        res.json(todos);
    }

    async createTodo(req, res) {
        const todo = new Todo(req.body);
        await todo.save();
        res.json(todo);
    }

    async updateTodo(req, res) {
        const todo = await Todo.findByIdAndUpdate(req.params.id, req.body, { new: true });
        res.json(todo);
    }

    async deleteTodo(req, res) {
        await Todo.findByIdAndRemove(req.params.id);
        res.json({ message: 'Todo deleted successfully' });
    }
}

module.exports = TodoController;
```

```typescript
// app.component.ts
import { Component, OnInit } from '@angular/core';
import { ChatService } from './chat.service';

@Component({
    selector: 'app-root',
    template: `
        <div>
            <input [(ngModel)]="message" />
            <button (click)="sendMessage()">Send</button>
            <ul>
                <li *ngFor="let msg of messages">{{ msg.message }} ({{ msg.sender }})</li>
            </ul>
        </div>
    `
})
export class AppComponent implements OnInit {
    message = '';
    messages = [];

    constructor(private chatService: ChatService) {}

    ngOnInit(): void {
        this.chatService.getMessages().subscribe((messages) => {
            this.messages = messages;
        });
    }

    sendMessage(): void {
        this.chatService.sendMessage({ message: this.message, sender: 'User' }).subscribe(() => {
            this.message = '';
        });
    }
}
```

### **chat.service.ts**
```typescript
// chat.service.ts
import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';

@Injectable({
    providedIn: 'root'
})
export class ChatService {
    private apiUrl = 'http://localhost:3000/api/chat';

    constructor(private http: HttpClient) {}

    sendMessage(message: any): any {
        return this.http.post(`${this.apiUrl}/messages`, message);
    }

    getMessages(): any {
        return this.http.get(`${this.apiUrl}/messages`);
    }
}
```

### **app.component.ts**
```typescript
// app.component.ts
import { Component, OnInit } from '@angular/core';
import { ChatService } from './chat.service';

@Component({
    selector: 'app-root',
    template: `
        <div>
            <input [(ngModel)]="message" />
            <button (click)="sendMessage()">Send</button>
            <ul>
                <li *ngFor="let msg of messages">{{ msg.message }} ({{ msg.sender }}) - {{ msg.timestamp }}</li>
            </ul>
        </div>
    `
})
export class AppComponent implements OnInit {
    message = '';
    messages = [];

    constructor(private chatService: ChatService) {}

    ngOnInit(): void {
        this.chatService.getMessages().subscribe((messages) => {
            this.messages = messages;
        });
    }

    sendMessage(): void {
        this.chatService.sendMessage({ message: this.message, sender: 'User' }).subscribe(() => {
            this.message = '';
            this.chatService.getMessages().subscribe((messages) => {
                this.messages = messages;
            });
        });
    }
}
```

### **chat.service.ts**
```typescript
// chat.service.ts
import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';

@Injectable({
    providedIn: 'root'
})
export class ChatService {
    private apiUrl = 'http://localhost:3000/api/chat';

    constructor(private http: HttpClient) {}

    sendMessage(message: any): any {
        return this.http.post(`${this.apiUrl}/messages`, message);
    }

    getMessages(): any {
        return this.http.get(`${this.apiUrl}/messages`);
    }
}
```

## Project Structure

*   **Backend (Node.js, Express, MongoDB)**
    *   `server.js`: Server-side logic
    *   `models/ChatMessage.js`: MongoDB model for chat messages
    *   `routes/chat.js`: API routes for chat functionality
*   **Frontend (Angular)**
    *   `app.module.ts`: Angular module
    *   `app.component.ts`: Angular component
    *   `chat.service.ts`: Service for interacting with backend API

## Backend (Node.js, Express, MongoDB)

### **server.js**
```javascript
// server.js
const express = require('express');
const app = express();
const mongoose = require('mongoose');
const bodyParser = require('body-parser');
const chatRoutes = require('./routes/chat');

mongoose.connect('mongodb://localhost:27017/chatapp', { useNewUrlParser: true, useUnifiedTopology: true });

app.use(bodyParser.json());
app.use('/api/chat', chatRoutes);

const port = 3000;
app.listen(port, () => {
    console.log(`Server listening on port ${port}`);
});
```

### **models/ChatMessage.js**
```javascript
// models/ChatMessage.js
const mongoose = require('mongoose');

const chatMessageSchema = new mongoose.Schema({
    message: String,
    sender: String,
    timestamp: Date
});

module.exports = mongoose.model('ChatMessage', chatMessageSchema);
```

### **routes/chat.js**
```javascript
// routes/chat.js
const express = require('express');
const router = express.Router();
const ChatMessage = require('../models/ChatMessage');

router.post('/messages', (req, res) => {
    const chatMessage = new ChatMessage({
        message: req.body.message,
        sender: req.body.sender,
        timestamp: new Date()
    });

    chatMessage.save((err) => {
        if (err) {
            res.status(500).send(err);
        } else {
            res.send({ message: 'Message sent successfully' });
        }
    });
});

router.get('/messages', (req, res) => {
    ChatMessage.find().then((messages) => {
        res.send(messages);
    }).catch((err) => {
        res.status(500).send(err);
    });
});

module.exports = router;
```
